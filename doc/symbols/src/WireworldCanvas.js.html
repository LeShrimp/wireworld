  5 /**
  6  *
  7  * @param {Wireworld} wireworld
  8  * @param {HTMLCanvasElement} htmlCanvasElement
  9  * @param {number} cellWidth
 10  * @constructor
 11  */
 12 var WireworldCanvas = function (wireworld, htmlCanvasElement, cellWidth) {
 13     /**  @type {HTMLCanvasElement} */
 14     this.htmlCanvasElement  = htmlCanvasElement;
 15     /**  @type {CanvasRenderingContext2D} */
 16     this.ctx                = this.htmlCanvasElement.getContext('2d');
 17     /** @type {number} */
 18     this.width              = htmlCanvasElement.width;
 19     /** @type {number} */
 20     this.height             = htmlCanvasElement.height;
 21     /** @type {Wireworld} */
 22     this.wireworld          = wireworld;
 23     /** @type {number} */
 24     this.cellWidth          = cellWidth;
 25 }
 26 
 27 /**
 28  *
 29  * @param width
 30  * @param height
 31  * @param id
 32  * @returns {HTMLElement}
 33  */
 34 WireworldCanvas.createCanvasElement = function(width, height, id) {
 35     var e = document.createElement('canvas');
 36     e.setAttribute('width', width);
 37     e.setAttribute('height', height);
 38     e.id = id;
 39 
 40     return e;
 41 }
 42 
 43 /**
 44  * Draws an instance of wireworld to the given Canvas.
 45  */
 46 WireworldCanvas.prototype.draw = function () {
 47     for (var i=0; i<this.wireworld.columns; i++) {
 48         for (var j=0; j<this.wireworld.rows; j++) {
 49             this.drawCell(i, j, this.wireworld.cells[i][j]);
 50         }
 51     }
 52 }
 53 
 54 
 55 /**
 56  *
 57  * @param i
 58  * @param j
 59  * @returns {{x: number, y: number, w: (number|*), h: *}}
 60  */
 61 WireworldCanvas.prototype.getCellRect = function (i, j) {
 62     return {
 63         x: i * this.cellWidth,
 64         y: j * this.cellWidth,
 65         w: this.cellWidth,
 66         h: this.cellWidth
 67     };
 68 }
 69 
 70 
 71 /**
 72  *
 73  * @param i
 74  * @param j
 75  * @param state
 76  */
 77 WireworldCanvas.prototype.drawCell = function (i, j, state) {
 78     var rect = this.getCellRect(i, j);
 79     this.ctx.beginPath();
 80     this.ctx.strokeStyle = 'black';
 81     switch (state) {
 82         case CircuitBoard.WW_COPPER:
 83             this.ctx.fillStyle = '#FF9900';
 84             break;
 85 
 86         case CircuitBoard.WW_EHEAD:
 87             this.ctx.fillStyle = '#000099';
 88             break;
 89 
 90         case CircuitBoard.WW_ETAIL:
 91             this.ctx.fillStyle = '#0099FF';
 92             break;
 93 
 94         case CircuitBoard.WW_BLACK:
 95         default:
 96             this.ctx.fillStyle = '#4C4747';
 97             break;
 98     }
 99     this.ctx.rect(rect.x+0.5, rect.y+0.5, rect.w, rect.h);
100     this.ctx.fill();
101     this.ctx.stroke();
102 }
