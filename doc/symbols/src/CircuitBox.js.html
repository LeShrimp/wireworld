  7 /**
  8  *
  9  * @constructor
 10  */
 11 var CircuitBox = function() {
 12     this.circuits = {};
 13 };
 14 
 15 /**
 16  *
 17  * @param {Wireworld} wireworld
 18  * @param {number} amount
 19  * @return {number} Id for the added circuit
 20  */
 21 CircuitBox.prototype.addCircuit = (function() {
 22     var id = 0;
 23     return function (wireworld, count) {
 24         id++;
 25         this.circuits[id] = {
 26             wireworld: wireworld,
 27             count: count
 28         }
 29         return id;
 30     }
 31 })();
 32 
 33 
 34 /**
 35  * Increases the amount available in the toolbox.
 36  * @param circuitId
 37  */
 38 CircuitBox.prototype.incCount = function (circuitId) {
 39     this.circuits[circuitId].count++;
 40 };
 41 
 42 
 43 /**
 44  * Decreases the amount of that circuit.
 45  * @param circuitId
 46  * @returns {boolean} false if amount was already 0, true otherwise.
 47  */
 48 CircuitBox.prototype.decCount = function (circuitId) {
 49     var circuit = this.circuits[circuitId];
 50 
 51     if (!circuit.count)
 52         return false;
 53 
 54     circuit.count--;
 55     return true;
 56 }
