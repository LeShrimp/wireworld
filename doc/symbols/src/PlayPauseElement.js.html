  6 /**
  7  *
  8  * @param htmlElement
  9  * @param onPlay
 10  * @param onPause
 11  * @constructor
 12  */
 13 var PlayPauseElement = function (htmlElement, onPlay, onPause) {
 14     this.htmlElement = htmlElement;
 15     this.isPlay = 0;
 16 
 17     this.htmlElement.addEventListener('click', function() {
 18         this.isPlay = !this.isPlay;
 19         this.isPlay ? onPlay() : onPause();
 20     });
 21 };
