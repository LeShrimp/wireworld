  7 /**
  8  *
  9  * @param {Wireworld} wireworld
 10  * @param {number} count
 11  * @param {number} blueprintId
 12  * @constructor
 13  */
 14 var Blueprint = function (wireworld, count, blueprintId) {
 15     this.wireworld = wireworld;
 16     this.count = count;
 17     this.id = blueprintId;
 18 };
 19 
 20 /**
 21  *
 22  * @constructor
 23  */
 24 var BlueprintBox = function() {
 25     this.blueprints = {};
 26 };
 27 
 28 
 29 /**
 30  *
 31  * @param {Wireworld} wireworld
 32  * @param {number} amount
 33  * @return {number} Id for the added circuit
 34  */
 35 BlueprintBox.prototype.addBlueprint = (function() {
 36     var id = 0;
 37     return function (wireworld, count) {
 38         id++;
 39         this.blueprints[id] = new Blueprint(wireworld, count, id);
 40         return id;
 41     }
 42 })();

 45 /**
 46  * Increases the amount available in the toolbox.
 47  * @param blueprint
 48  */
 49 BlueprintBox.prototype.incCount = function (blueprint) {
 50     blueprint.count++;
 51 };
 52 
 53 
 54 /**
 55  * Decreases the amount of that circuit.
 56  * @param blueprint
 57  * @returns {boolean} false if amount was already 0, true otherwise.
 58  */
 59 BlueprintBox.prototype.decCount = function (blueprint) {
 60 
 61     if (!blueprint.count)
 62         return false;
 63 
 64     blueprint.count--;
 65     return true;
 66 }
