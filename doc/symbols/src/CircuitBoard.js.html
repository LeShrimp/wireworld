  8 /**
  9  *
 10  * @param {number} i
 11  * @param {number} j
 12  * @param {Blueprint} blueprint The blueprint this circuit is made of.
 13  * @param {number} id
 14  * @constructor
 15  */
 16 var Circuit = function(i, j, blueprint, id) {
 17     this.i = i;
 18     this.j = j;
 19     this.blueprint = blueprint;
 20     this.id = id;
 21 };
 22 
 23 /**
 24  * Creates a new instance of a Wireworld Circuitboard. This is an extension of
 25  * Wireworld, accepting a new cell state
 26  *
 27  * @augments Wireworld
 28  * @this {CircuitBoard}
 29  * @params {Array} cells A 2-dimensional rectangular array, giving initial values of the cells
 30  * @constructor
 31  */
 32 var CircuitBoard = function(cells) {
 33     Wireworld.call(this, cells);
 34     this.circuits = {};
 35 };
 36 
 37 copyProperties(CircuitBoard, Wireworld); //Copy static properties
 38 copyPrototype(CircuitBoard, Wireworld);
 39 
 40 CircuitBoard.WW_EMPTY = 4;
 41 
 42 /**
 43  * Places circuit at given position.
 44  *
 45  * @param {number} i
 46  * @param {number} j
 47  * @param {Blueprint} blueprint
 48  * @return {number} Id of the placed circuit
 49  */
 50 CircuitBoard.prototype.placeCircuit = (function () {
 51     var id=1;
 52     return function (i, j, blueprint) {
 53         id++;
 54         this.circuits[id] = new Circuit(i, j, blueprint, id);
 55         return id;
 56     }
 57 })();
 58 
 59 
 60 /**
 61  * If there is a circuit covering cell (i,j) return its id.
 62  *
 63  * @param i
 64  * @param j
 65  * @return Circuit id if there is any circuit, null otherwise
 66  */
 67 CircuitBoard.prototype.getCircuitAtPos = function (i, j) {
 68     for (var id in this.circuits) {
 69         var circuit = this.circuits[id];
 70         if (circuit.i <= i
 71             && circuit.i + circuit.blueprint.wireworld.columns > i
 72             && circuit.j <= j
 73             && circuit.j + circuit.blueprint.wireworld.rows > j) {
 74                 return circuit;
 75         }
 76     }
 77     return null;
 78 }

 81 /**
 82  * Remove circuit with given id.
 83  *
 84  * @param {Circuit} circuit
 85  */
 86 CircuitBoard.prototype.removeCircuit = function (circuit) {
 87     delete this.circuits[circuit.id];
 88 }

 91 /**
 92  *
 93  * @param i
 94  * @param j
 95  * @param {Blueprint} blueprint
 96  */
 97 CircuitBoard.prototype.isPlacementLegal = function (i, j, blueprint) {
 98     var k, l;
 99     var wireworld = blueprint.wireworld;
100 
101     if (i<0 || j<0 || i+wireworld.columns > this.columns || j+wireworld.rows > this.rows) {
102         return false;
103     }
104 
105     for (k=0; k<wireworld.columns; k++) {
106         for (l=0; l<wireworld.rows; l++) {
107             if (this.cells[i+k][j+l] != CircuitBoard.WW_EMPTY) {
108                 return false;
109             }
110         }
111     }
112     for (k=0; k<wireworld.columns; k++) {
113         for (l=0; l<wireworld.rows; l++) {
114             if (this.getCircuitAtPos(i+k, j+l) != null) {
115                 return false;
116             }
117         }
118     }
119     return true;
120 }
