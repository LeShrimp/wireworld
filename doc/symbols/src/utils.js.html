  6 function isArray(o) {
  7     return Object.prototype.toString.call(o) === '[object Array]';
  8 }
  9 
 10 /**
 11  * Checks wether given array is rectangular. And of given width and height
 12  * (if provided). First dimension ist considered as width, second as height.
 13  * Array is supposed to be of the form a[x][y], x in width direction. y in
 14  * height direction.
 15  *
 16  * @param {Array} a
 17  * @param {number} [width]
 18  * @param {number} [height]
 19  * @returns {boolean}
 20  */
 21 function isRectArray(a, width, height) {
 22     if (!isArray(a)) {
 23         return false;
 24     }
 25     if (typeof(width)!=='undefined' && a.length != width) {
 26         return false;
 27     }
 28     for (var i=0; i<a.length; i++) {
 29         if (!isArray(a[i])) {
 30             return false;
 31         }
 32         if (typeof(height)!=='undefined' && a[i].length != height) {
 33             return false;
 34         }
 35     }
 36     return true;
 37 }

 40 function transpose(a) {
 41     if (!isRectArray(a)) {
 42         throw 'Unable to transpose argument. Invalid format.';
 43     }
 44     at = [];
 45     for (var j=0; j<a[0].length; j++) {
 46         at[j] = [];
 47         for (var i=0; i<a.length; i++) {
 48             at[j][i] = a[i][j];
 49         }
 50     }
 51 
 52     return at;
 53 }

 56 /**
 57  * Copys every enumerable property of the second object to
 58  * the first object.
 59  *
 60  * @param {Object} obj1
 61  * @param {Object} obj2
 62  */
 63 function copyProperties(obj1, obj2) {
 64     for (var prop in obj2) {
 65         obj1[prop] = obj2[prop];
 66     }
 67 }

 70 /**
 71  * Copys every enumerable property of the second arguments prototype to
 72  * the prototype of the first argument.
 73  *
 74  * @param {function} constr1
 75  * @param {function} constr2
 76  */
 77 function copyPrototype(constr1, constr2) {
 78     proto1 = constr1.prototype;
 79     proto2 = constr2.prototype;
 80     for (var prop in proto2) {
 81         proto1[prop] = proto2[prop];
 82     }
 83 }

 86 /**
 87  * Primitive implementation of each. Executes callback(a[i])
 88  * for every element in the array a.
 89  *
 90  * @param {Array} a
 91  * @param {function} callback
 92  */
 93 function forEach(a, callback) {
 94     for (var i in a) {
 95         callback(a[i]);
 96     }
 97 }
